1. Cuando asignamos una function a una variable

window.x=10
window.foo={
    x:11,
    write:function(){
        console.log(this.x)
    }
}

window.foo.write() //11
window.write = window.foo.write.bind(window.foo);
window.write() //10

2. Cuando pasamos una function a otra function como argumento

class Bar{
    constructor(writer){
        writer();  // undefined
        this.writer = writer
        this.writer(); //Bar
    }
}

class Foo{
  constructor(){
    //new Bar(this.writer.bind(this)) (1)
    new Bar(()=>this.writer()) //No(2)
  }
  writer(){
    console.log(this)
  }   
}
